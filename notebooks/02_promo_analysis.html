# Chapter 4 Part A: Promotion / Descriptive Results

from pathlib import Path
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# ------------------------------------------------------------
# Paths
# ------------------------------------------------------------

PROJECT_ROOT = Path.cwd().resolve().parent if Path.cwd().name == "notebooks" else Path.cwd()
RAW_DIR = PROJECT_ROOT / "data" / "raw"
RESULT_DIR = PROJECT_ROOT / "data" / "processed" / "chapter4_results"
GRAPH_DIR = PROJECT_ROOT / "data" / "processed" / "report_graphs"

RESULT_DIR.mkdir(parents=True, exist_ok=True)
GRAPH_DIR.mkdir(parents=True, exist_ok=True)

def save_and_show(name):
    path = GRAPH_DIR / name
    plt.tight_layout()
    plt.savefig(path, dpi=300, bbox_inches="tight")
    plt.show()
    plt.close()
    print(f"Saved: {path}")

# ------------------------------------------------------------
# Load orders data
# ------------------------------------------------------------

if "orders_df" in globals():
    df = orders_df.copy()
    print("Using existing orders_df.")
else:
    possible_files = [
        RAW_DIR / "JD_order_data.csv",
        RAW_DIR / "order_data.csv",
        RAW_DIR / "orders.csv",
        RAW_DIR / "JD_orders.csv"
    ]

    order_path = next((p for p in possible_files if p.exists()), None)

    if order_path is None:
        raise FileNotFoundError(
            "Order file not found. Please check your raw data folder and file name."
        )

    df = pd.read_csv(order_path)
    print(f"Loaded order data from: {order_path}")

# ------------------------------------------------------------
# Standardize column names
# ------------------------------------------------------------

df.columns = [c.strip() for c in df.columns]

if "order_ID" in df.columns and "order_id" not in df.columns:
    df = df.rename(columns={"order_ID": "order_id"})

# ------------------------------------------------------------
# Create discount variables
# ------------------------------------------------------------

if "discount_rate" not in df.columns:
    if {"original_unit_price", "final_unit_price"}.issubset(df.columns):
        df["discount_rate"] = (
            df["original_unit_price"] - df["final_unit_price"]
        ) / df["original_unit_price"]

        df["discount_rate"] = (
            df["discount_rate"]
            .replace([np.inf, -np.inf], np.nan)
            .fillna(0)
        )

        df["discount_rate"] = df["discount_rate"].clip(lower=0)
    else:
        print("Price columns not found. discount_rate cannot be created.")

if "is_discounted" not in df.columns and "discount_rate" in df.columns:
    df["is_discounted"] = (df["discount_rate"] > 0).astype(int)

if "total_discount_amount" not in df.columns:
    if {"original_unit_price", "final_unit_price", "quantity"}.issubset(df.columns):
        df["total_discount_amount"] = (
            df["original_unit_price"] - df["final_unit_price"]
        ) * df["quantity"]

        df["total_discount_amount"] = df["total_discount_amount"].clip(lower=0)

# ------------------------------------------------------------
# 1. Descriptive statistics table
# ------------------------------------------------------------

desc_cols = [
    c for c in [
        "quantity",
        "original_unit_price",
        "final_unit_price",
        "discount_rate",
        "total_discount_amount"
    ]
    if c in df.columns
]

descriptive_table = df[desc_cols].describe().T
descriptive_table.to_csv(RESULT_DIR / "chapter4_descriptive_statistics.csv")

print("\nDescriptive Statistics:")
display(descriptive_table)

# ------------------------------------------------------------
# 2. Discounted vs Non-discounted Summary
# ------------------------------------------------------------

if {"is_discounted", "quantity"}.issubset(df.columns):

    discount_summary = (
        df.groupby("is_discounted")
        .agg(
            orders=("quantity", "count"),
            avg_quantity=("quantity", "mean"),
            total_units=("quantity", "sum")
        )
        .reset_index()
    )

    discount_summary["discount_status"] = discount_summary["is_discounted"].map({
        0: "Non-discounted",
        1: "Discounted"
    })

    discount_summary.to_csv(
        RESULT_DIR / "chapter4_discount_vs_quantity.csv",
        index=False
    )

    print("\nDiscounted vs Non-discounted Orders:")
    display(discount_summary)

    plt.figure(figsize=(7, 5))
    plt.bar(
        discount_summary["discount_status"],
        discount_summary["avg_quantity"]
    )
    plt.ylabel("Average Quantity per Order Line")
    plt.title("Average Quantity: Discounted vs Non-discounted Orders")

    for i, v in enumerate(discount_summary["avg_quantity"]):
        plt.text(i, v, f"{v:.2f}", ha="center", va="bottom")

    save_and_show("chapter4_graph1_discount_vs_quantity.png")

else:
    print("Skipped discounted vs non-discounted graph because required columns are missing.")

# ------------------------------------------------------------
# 3. Discount Bucket Summary
# ------------------------------------------------------------

if {"discount_rate", "quantity"}.issubset(df.columns):

    df["discount_bucket"] = pd.cut(
        df["discount_rate"],
        bins=[-0.001, 0, 0.05, 0.10, 0.20, 0.40, 1.00],
        labels=["0%", "0-5%", "5-10%", "10-20%", "20-40%", "40%+"]
    )

    bucket_summary = (
        df.groupby("discount_bucket", observed=False)
        .agg(
            orders=("quantity", "count"),
            avg_quantity=("quantity", "mean"),
            total_units=("quantity", "sum")
        )
        .reset_index()
    )

    bucket_summary.to_csv(
        RESULT_DIR / "chapter4_discount_bucket_summary.csv",
        index=False
    )

    print("\nDiscount Bucket Summary:")
    display(bucket_summary)

    plt.figure(figsize=(8, 5))
    plt.bar(
        bucket_summary["discount_bucket"].astype(str),
        bucket_summary["avg_quantity"]
    )
    plt.xlabel("Discount Rate Bucket")
    plt.ylabel("Average Quantity per Order Line")
    plt.title("Average Quantity by Discount Rate Bucket")

    for i, v in enumerate(bucket_summary["avg_quantity"]):
        if pd.notna(v):
            plt.text(i, v, f"{v:.2f}", ha="center", va="bottom")

    save_and_show("chapter4_graph2_avg_quantity_by_discount_bucket.png")

else:
    print("Skipped discount bucket graph because required columns are missing.")

print("\nChapter 4 Part A completed.")
print(f"Tables saved to: {RESULT_DIR}")
print(f"Graphs saved to: {GRAPH_DIR}")
